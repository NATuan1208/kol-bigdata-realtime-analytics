def cache_to_redis(df: DataFrame):
    """Cache scores to Redis for fast lookup"""
    import redis
    
    r = redis.Redis(host="redis", port=6379)
    
    rows = df.collect()
    pipe = r.pipeline()
    
    for row in rows:
        key = f"kol:score:{row['kol_id']}"
        value = json.dumps({
            "trust_score": row["trust_score"],
            "trust_label": row["trust_label"],
            "timestamp": row["timestamp"],
        })
        pipe.setex(key, 3600, value)  # TTL 1 hour
    
    pipe.execute()  # Atomic batch write

### 6.2 Cache Client (`client.py`)

```python
class CacheClient:
    """
    Async Redis cache client with connection pooling.
    Thread-safe and designed for high-concurrency workloads.
    """
    
    # Key methods:
    
    async def set_kol_profile(self, profile: Dict) -> bool:
        """Store KOL profile as hash."""
        key = CacheKeys.kol_profile(profile["kol_id"])
        return await self.hset_with_ttl(key, profile, TTL.KOL_PROFILE)
    
    async def get_kol_profile(self, kol_id: str) -> Optional[Dict]:
        """Get KOL profile from cache."""
        key = CacheKeys.kol_profile(kol_id)
        return await self.hgetall(key) or None
    
    async def set_top_kols_by_followers(
        self, 
        platform: str, 
        kols: List[Dict],
        limit: int = 100
    ) -> bool:
        """Store top KOLs ranking as sorted set."""
        key = CacheKeys.top_kols_by_followers(platform, limit)
        mapping = {kol["kol_id"]: float(kol["followers_count"]) for kol in kols}
        return await self.zadd(key, mapping, TTL.TOP_FOLLOWERS)
    
    async def set_platform_stats(self, stats: Dict) -> bool:
        """Store platform statistics."""
        key = CacheKeys.platform_stats()
        return await self.hset_with_ttl(key, stats, TTL.PLATFORM_STATS)
```

### 6.3 Cache Warmer (`cache_warmer.py`)

```python
class CacheWarmer:
    """
    Cache warming utility.
    Loads data from Trino (Silver layer) into Redis cache.
    """
    
    async def warm_all(self):
        """Full cache warm."""
        await self.warm_profiles()      # Hot KOL profiles
        await self.warm_top_kols()      # Rankings per platform
        await self.warm_trending_content()  # Trending content
        await self.warm_platform_stats()    # Platform statistics
    
    async def warm_top_kols(self):
        """Warm top KOLs rankings."""
        for platform in ["tiktok", "twitter"]:
            kols = self.trino.execute_dict(f"""
                SELECT kol_id, followers_count
                FROM kol_profiles
                WHERE platform = '{platform}'
                ORDER BY followers_count DESC
                LIMIT 100
            """)
            await self._redis.set_top_kols_by_followers(platform, kols)
```
